    实例¶

    from utils import *
import net
import time
p=net.Proxy("127.0.0.1:2077") # 监听本机2077端口
p.start() # 启用代理
while True:
    req=p.request() # 获取待处理请求
    if req!=None: # 如果取得请求
        req.release() # 释放请求
    resp=p.response() # 获取待处理响应
    if resp!=None: # 如果取得响应
        echo(resp.headers()) # 输出响应头
        resp.status_code(404) # 修改状态码为404
        resp.release() # 释放响应
    time.sleep(0.1) # 记得使用sleep避免cpu占用过高
